%% Usage:
%% This *.sty file adds Meta-Data to the PDF generated from LaTex.
%%/////////////////////////////////////////////////////////////////////////////
\newif\ifAddMetaData

\AddMetaDatatrue
%===============================
%  *** Add PDF meta data: ***
%===============================
\ifAddMetaData
    \makeatletter % changes the catcode of @ to 11
    \hypersetup{  %Add document properties in PDF using LaTeX:
        pdftitle= {PDF file}, %\@title,%
        pdfauthor= {Behzad Nouri}, %\@author,%
        pdfsubject={Technical},%
        pdfkeywords={},%
        pdfproducer={LaTeX},%
        pdfcreator={pdfLaTeX}
    }
    \makeatother % changes the catcode of @ back to 12
\fi
